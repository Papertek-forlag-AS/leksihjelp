1. Create new file `extension/content/spell-rules/nb-compound-gender.js` following the IIFE pattern from nb-gender.js:

   Structure:
   ```javascript
   (function () {
     'use strict';
     const host = typeof self !== 'undefined' ? self : globalThis;
     host.__lexiSpellRules = host.__lexiSpellRules || [];
     const { matchCase, escapeHtml } = host.__lexiSpellCore || {};

     const rule = {
       id: 'nb-compound-gender',
       lang: ['nb', 'nn'],
       priority: 71,  // Same tier as de-compound-gender, after nb-gender (10)
       severity: 'warning',

       explain: (finding) => {
         // Return { nb: string, nn: string } with student-friendly explanation
         // Use finding.actualGenus and finding.expectedGenus for specific message
         // Example NB: "Substantivet «fotballsko» er hankjønn. Bruk «en» i staden for «et»."
         // Example NN: "Substantivet «fotballsko» er hankjønn. Bruk «ein» i staden for «eit»."
       },

       check(ctx) {
         const { tokens, vocab, cursorPos, suppressed } = ctx;
         const nounGenus = vocab.nounGenus || new Map();
         const decompose = vocab.decomposeCompound;
         if (!decompose) return [];
         const out = [];

         // NB articles: en (m), ei (f), et (n)
         // NN articles: ein (m), ei (f), eit (n)
         const ARTICLE_GENDER = {
           'en': 'm', 'ei': 'f', 'et': 'n',    // NB
           'ein': 'm', 'eit': 'n',              // NN (ei = f, shared)
         };

         for (let i = 1; i < tokens.length; i++) {
           const t = tokens[i];
           const prev = tokens[i - 1];
           if (cursorPos != null && cursorPos >= t.start && cursorPos <= t.end + 1) continue;
           if (suppressed && suppressed.has(i)) continue;

           const articleGender = ARTICLE_GENDER[prev.word];
           if (!articleGender) continue;

           // Only fire for nouns NOT in nounGenus (known nouns handled by nb-gender at priority 10)
           if (nounGenus.has(t.word)) continue;

           const decomposition = decompose(t.word);
           if (!decomposition || decomposition.confidence !== 'high') continue;

           const inferredGender = decomposition.gender;
           if (!inferredGender || inferredGender === articleGender) continue;

           // Gender mismatch — flag it
           const correctArticle = /* look up correct article for inferredGender */;
           out.push({
             rule_id: rule.id,
             priority: rule.priority,
             severity: rule.severity,
             tokenIndex: i - 1,
             start: prev.start,
             end: t.end,
             original: prev.display + ' ' + t.display,
             fix: matchCase(prev.display, correctArticle) + ' ' + t.display,
             message: `Kjønnsfeil: «${escapeHtml(t.display)}» er ${genusLabel} — bruk «${correctArticle}»`,
             actualGenus: inferredGender,
             expectedGenus: articleGender,
           });
         }
         return out;
       },
     };

     host.__lexiSpellRules.push(rule);
     if (typeof module !== 'undefined' && module.exports) module.exports = rule;
   })();
   ```

   Key design points:
   - Article-to-gender mapping covers both NB and NN articles
   - Correct-article lookup: 'm' -> 'en'/'ein', 'f' -> 'ei', 'n' -> 'et'/'eit' (detect NB vs NN from ctx.lang if available, or default to NB articles)
   - explain() must return `{ nb: string, nn: string }` (check-explain-contract requirement)
   - Include `actualGenus` and `expectedGenus` on finding so explain() can generate specific messages
