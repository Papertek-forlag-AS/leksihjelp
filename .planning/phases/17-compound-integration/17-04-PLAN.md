1. In `extension/background/service-worker.js`, add a new message handler case inside the existing `chrome.runtime.onMessage.addListener` callback. The handler responds to `{type: 'SEND_REPORT', data: {...}}`:

```javascript
if (msg.type === 'SEND_REPORT') {
  (async () => {
    try {
      const stored = await chrome.storage.local.get(['sessionToken', 'siteUrl']);
      const base = stored.siteUrl || 'https://leksihjelp.no';
      const headers = { 'Content-Type': 'application/json', 'X-Lexi-Client': 'lexi-extension' };
      if (stored.sessionToken) headers['Authorization'] = 'Bearer ' + stored.sessionToken;
      const resp = await fetch(base + '/api/report', { method: 'POST', headers, body: JSON.stringify(msg.data) });
      sendResponse(resp.ok);
    } catch (_) { sendResponse(false); }
  })();
  return true; // async sendResponse
}
```

2. In `extension/content/spell-check.js`, replace the `sendReport` function (lines ~796-805) with a delegation to the service-worker:

```javascript
async function sendReport(data) {
  try {
    return await new Promise(resolve => {
      chrome.runtime.sendMessage({ type: 'SEND_REPORT', data }, ok => {
        resolve(ok ?? false);
      });
    });
  } catch (_) { return false; }
}
```

This removes the `fetch()` and `https://` URL from the scanned offline surface while preserving identical functionality.
