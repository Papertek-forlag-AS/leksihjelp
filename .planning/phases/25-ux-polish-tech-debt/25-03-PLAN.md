     if (e.key !== 'Tab' || !markers.length) return;
     if (activePopoverIdx < 0) return; // only when our popover is the active context

     const dir = e.shiftKey ? -1 : 1;
     const next = (activePopoverIdx + dir + markers.length) % markers.length;
     activePopoverIdx = next;
     markers[next].el.scrollIntoView({ behavior: 'smooth', block: 'center' });
     positionPopover(markers[next].rect);
