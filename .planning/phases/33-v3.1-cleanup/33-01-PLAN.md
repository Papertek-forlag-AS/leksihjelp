```javascript
async function populateDictState(viewState, lang, host) {
  const raw = await host.__lexiVocabStore?.getCachedLanguage?.(lang);
  if (!raw) return false;
  const builder = getBuilder();
  if (!builder) return false;
  // Determine NB sister + load it (uiLang === 'nn' uses 'nn' sister, else 'nb')
  const uiLang = host.__lexiI18n?.getUiLanguage?.() || 'nb';
  const noLang = uiLang === 'nn' ? 'nn' : 'nb';
  let sisterRaw = null;
  if (lang !== noLang) {
    try { sisterRaw = await host.__lexiVocabStore?.getCachedLanguage?.(noLang); } catch {}
  }
  const state = builder.buildDictState({
    raw, sisterRaw, lang, noLang,
    vocabCore: host.__lexiVocabCore,
  });
  viewState.dictionary = raw;
  viewState.noDictionary = sisterRaw;
  Object.assign(viewState, state);   // allWords, inflectionIndex, nounGenusMap, ...
  return true;
}
```
