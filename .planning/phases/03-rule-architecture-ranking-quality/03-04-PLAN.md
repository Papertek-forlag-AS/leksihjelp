```javascript
// Effective frequency: Zipf sidecar (NB/NN) takes priority; seam-normalized
// entry.zipf is the fallback + sole source for DE/ES/FR/EN.
function getEffectiveFreq(entry) {
  const z = VOCAB.getFrequency(entry.word);
  if (typeof z === 'number') return z;
  if (typeof entry.zipf === 'number') return entry.zipf;
  return 0;
}

// Shared-suffix length helper for tiebreaker — mirrors the spell-check core
// helper but local (word-prediction is disallowed from importing from the
// spell-check surface, per INFRA-04 structural separability).
function sharedSuffixLen(a, b) {
  if (!a || !b) return 0;
  const la = a.length, lb = b.length;
  const n = Math.min(la, lb);
  let i = 0;
  while (i < n && a[la - 1 - i] === b[lb - 1 - i]) i++;
  return i;
}
```

Step 2 — Refactor `applyBoosts`. Current body has ~12 if-branches mixing per-language and general signals. Convert to:

```javascript
function freqSignal(entry, ctx) {
  // Pitfall 7: never use raw entry.frequency — always getEffectiveFreq.
  // Multiplier tuned so top-Zipf (~7) adds 140, bottom-Zipf (0) adds 0.
  // Bigram signal (line ~1007 today) caps at +120 — freq signal tops out
  // comparable, so neither dominates.
  const z = getEffectiveFreq(entry);
  return z * 20;
}

function lowFreqDemotion(entry, ctx) {
  // WP-04: demote rare/obscure forms so the top-3 isn't dominated by random
  // high-score-but-low-frequency artefacts. Floor at ~Zipf 1.5 (≈ occurs
  // 30+ times per million) — generous floor so learner-core vocab never
  // triggers it.
  const z = getEffectiveFreq(entry);
  if (z > 0 && z < 1.5) return -80;
  return 0;
}

function applyBoosts(entry, score, scored, pronounContext, hasModalVerb, detectedTense, query, expectedPOS, genderContext, posStrength, caseContext, previousWord, previousTwoWords, numberContext = null, definitenessContext = null) {
  const ctx = {
    lang: VOCAB.getLanguage(),
    pronounContext, hasModalVerb, detectedTense, query,
    expectedPOS, genderContext, posStrength, caseContext,
    previousWord, previousTwoWords, numberContext, definitenessContext,
  };

  // 1. Typo-threshold (existing — keep as first signal because it can mutate `entry`)
  let workingEntry = entry;
  if (entry.type === 'typo' && score >= 50) {
    if (query && query.length / entry.display.length >= 0.6) {
      score += 150;
    } else {
      workingEntry = { ...entry, type: 'base' };
      score += 30;
    }
  }

  // 2. Recency
  if (recentWordsSet.has(workingEntry.word)) score += 50;

  // 3. Modal + infinitive (existing, unchanged)
  if (hasModalVerb && workingEntry.type === 'base' && workingEntry.bank === 'verbbank') score += 250;
  if (hasModalVerb && workingEntry.type === 'conjugation' && workingEntry.formKey === 'infinitiv') score += 250;

  // 4. Pronoun + tense (existing, unchanged — NB/NN guarded internally)
  if (!hasModalVerb && pronounContext && workingEntry.type === 'conjugation' && workingEntry.pronoun === pronounContext) score += 200;
  if (!hasModalVerb && pronounContext === '_nb_pronoun') {
    const targetTense = detectedTense || 'present';
    if (workingEntry.type === 'base' && workingEntry.bank === 'verbbank') score += 150;
    if (workingEntry.type === 'conjugation' && workingEntry.tenseKey === targetTense) score += 200;
  }
  if (detectedTense && workingEntry.type === 'conjugation' && workingEntry.tenseKey === detectedTense) score += 180;

  // 5. POS expectation (existing, unchanged)
  if (expectedPOS === 'noun_adj') {
    const isNounOrAdj = workingEntry.bank === 'nounbank' || workingEntry.bank === 'adjectivebank' ||
      workingEntry.type === 'nounform' || workingEntry.type === 'plural' || workingEntry.type === 'case' ||
      workingEntry.type === 'comparative' || workingEntry.type === 'superlative' || workingEntry.type === 'adjform';
    const isVerb = workingEntry.bank === 'verbbank' || workingEntry.type === 'conjugation';
    if (isNounOrAdj) score += posStrength >= 2 ? 150 : 100;
    else if (isVerb && posStrength >= 2) score -= 100;
  }

  // 6. Gender agreement (existing, unchanged)
  if (genderContext && workingEntry.genus === genderContext) score += 120;

  // 7. NB/NN number + definiteness (existing, unchanged — NB/NN language is
  //    not checked here because nounform/adjform + number/definiteness are
  //    NB/NN-specific fields in practice; other languages' entries won't
  //    have both `number` and `definiteness`)
  if ((numberContext || definitenessContext) && (workingEntry.type === 'nounform' || workingEntry.type === 'adjform')) {
    const numMatch = numberContext && workingEntry.number === numberContext;
    const defMatch = definitenessContext && workingEntry.definiteness === definitenessContext;
    const adjFlertall = workingEntry.type === 'adjform' && workingEntry.number === 'flertall' && workingEntry.definiteness == null;
    if (numMatch && (defMatch || !definitenessContext || adjFlertall)) score += 130;
    else if (numMatch || defMatch) score += 60;
    else {
      const numConflict = numberContext && workingEntry.number && workingEntry.number !== numberContext;
      const defConflict = definitenessContext && workingEntry.definiteness && workingEntry.definiteness !== definitenessContext;
      if (numConflict || defConflict) score -= 80;
    }
  }

  // 8. DE case (existing, unchanged — self-guards via workingEntry.caseName presence)
  if (caseContext && workingEntry.type === 'case' && workingEntry.caseName) {
    if (caseContext === 'wechsel') {
      if (workingEntry.caseName === 'akkusativ' || workingEntry.caseName === 'dativ') score += 80;
    } else if (workingEntry.caseName === caseContext) score += 150;
  }

  // 9. Bigrams (existing, unchanged)
  const bigramData = VOCAB.getBigrams();
  if (bigramData) {
    let bigramWeight = 0;
    if (previousTwoWords) {
      const pairs2 = bigramData[previousTwoWords];
      if (pairs2) bigramWeight = pairs2[workingEntry.word] || 0;
    }
    if (!bigramWeight && previousWord) {
      const pairs1 = bigramData[previousWord];
      if (pairs1) bigramWeight = pairs1[workingEntry.word] || 0;
    }
    if (bigramWeight > 0) score += bigramWeight * 40;
  }

  // 10. NEW — Frequency signal (WP-01). Zipf sidecar for NB/NN; seam-normalized
  //     entry.zipf for DE/ES/FR/EN.
  score += freqSignal(workingEntry, ctx);

  // 11. NEW — Low-frequency demotion (WP-04).
  score += lowFreqDemotion(workingEntry, ctx);

  scored.push({ ...workingEntry, score });
}
```
