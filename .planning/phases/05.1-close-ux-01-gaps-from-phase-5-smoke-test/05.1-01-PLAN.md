Step 1 — Create `scripts/validate-adj-declensions.js` as an ES module (papertek-vocabulary uses `"type": "module"`). Shape inspired by `scripts/audit-nn-adjectivebank.js`. Use this skeleton verbatim:

```javascript
import { readFileSync } from 'fs';
import { join, dirname } from 'path';
import { fileURLToPath } from 'url';

const __dirname = dirname(fileURLToPath(import.meta.url));
const ROOT = join(__dirname, '..');
const BANKS = [
  ['nb', join(ROOT, 'vocabulary/lexicon/nb/adjectivebank.json')],
  ['nn', join(ROOT, 'vocabulary/lexicon/nn/adjectivebank.json')],
];

// Known irregulars — whitelist. Add as audit surfaces cases.
// god_adj → godt (regular -t after d), liten → lite (mono irregular), gammal → gammalt
const IRREGULAR = new Set([
  'liten_adj', 'god_adj', 'gammal_adj', 'gamal_adj', 'annen_adj',
  'all_adj', 'egen_adj', 'egen_adj', 'sann_adj',
]);

// Monosyllabic vowel-ending root — ends in a vowel and is ≤ 3 chars.
// Covers: blå, ny, grå, rå, fri, sky, tro, må, få, tø
const MONOSYLLABIC_VOWEL = /^[a-zæøå]{1,3}[aeiouyæøå]$/;

function predictNeuter(root) {
  const r = root.toLowerCase();
  if (r.endsWith('e')) return r;                              // blide → blide
  if (r.endsWith('ig')) return r + 't';                        // viktig → viktigt
  if (r.endsWith('sk')) return r;                              // norsk → norsk
  if (r.endsWith('t')) return r;                               // kort → kort
  if (/(dd|tt|nn|mm|ll|ss|kk|pp|gg|bb)$/.test(r)) return r;    // double consonant
  if (MONOSYLLABIC_VOWEL.test(r)) return r + 'tt';             // blå → blått
  return r + 't';                                              // default: pen → pent
}

let defects = 0;
const defectList = [];
for (const [lang, path] of BANKS) {
  let bank;
  try { bank = JSON.parse(readFileSync(path, 'utf8')); } catch (e) {
    console.error(`[${lang}] cannot read ${path}: ${e.message}`);
    process.exit(1);
  }
  for (const [id, entry] of Object.entries(bank)) {
    if (id.startsWith('_')) continue;          // skip _metadata
    if (IRREGULAR.has(id)) continue;
    const positiv = entry.declension?.positiv;
    if (!positiv) {
      defectList.push({ lang, id, kind: 'missing positiv' });
      defects++;
      continue;
    }
    if (!entry.word) continue;
    const expected = predictNeuter(entry.word);
    if (positiv.noytrum !== expected) {
      defectList.push({ lang, id, kind: 'wrong neuter', word: entry.word, got: positiv.noytrum, want: expected });
      defects++;
    }
  }
}

// Pretty report
if (defects === 0) {
  console.log(`OK — NB + NN adjectivebanks pass declension audit (${BANKS.length} banks checked)`);
  process.exit(0);
}
console.error(`FAIL — ${defects} defect(s) across NB + NN adjectivebanks`);
for (const d of defectList) {
  if (d.kind === 'wrong neuter') {
    console.error(`  [${d.lang}] ${d.id}: word="${d.word}" noytrum="${d.got}" expected="${d.want}"`);
  } else {
    console.error(`  [${d.lang}] ${d.id}: ${d.kind}`);
  }
}
process.exit(1);
```

Regex breakdown: `^[bcdfghjklmnpqrstvwxz]?[aeiouyæøå]$` matches:
- optional single leading consonant (not a vowel — prevents `aa` style from matching) then
- exactly one vowel at end-of-string.
- Total length ≤ 2 characters.

This catches `blå` (b+å), `ny` (n+y), `grå` (g+r+å — NO, 3 chars — does NOT match the 2-char limit!).

Per research's Pitfall 2, use the alternate regex from the Code Examples block that allows ≤3 chars with vowel-final:

```javascript
} else if (w.length <= 3 && /[aeiouyæøå]$/.test(w)) {
```

Trade-off: the 3-char form is looser but covers `grå` (3 chars), `fri` (3), `tro` (3), `sky` (3). Downside: a hypothetical 3-char consonant-cluster-vowel adjective like `nei` (2) or hypothetical `bra` (3) — both legitimately take `-tt` (`brått`). Use the length-based form.
