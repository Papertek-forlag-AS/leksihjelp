**A. Create `leksihjelp-sidepanel-host.js`** (lockdown-only file; lives in lockdown's repo, not in synced output).

```js
/**
 * leksihjelp-sidepanel-host.js — Phase 30, lockdown-only host file.
 *
 * Mounts the synced leksihjelp popup view modules into the writing-environment
 * resource panel's "Leksihjelp ordbok" tab. Deliberately omits surfaces that
 * don't belong in a school exam context: no audio, no auth, no payments,
 * no exam-toggle, no skriv/pin/vocab-refresh, no UI-language picker, no
 * spellcheck-alternates toggle.
 *
 * Source of truth for view logic lives upstream at:
 *   ../../leksihjelp/extension/popup/views/{dictionary,settings}-view.js
 *
 * This file declares only the inclusion contract — what we mount, with what
 * deps. Bug fixes to the views themselves go upstream.
 */
(function () {
    'use strict';

    function makeStorageAdapter(host) {
        return {
            get: (key) => new Promise(resolve => host.chrome.storage.local.get(key, resolve)),
            set: (obj) => new Promise(resolve => host.chrome.storage.local.set(obj, resolve)),
        };
    }

    function makeRuntimeAdapter(host) {
        return {
            sendMessage: (msg) => host.chrome.runtime.sendMessage(msg),
        };
    }

    function makeI18n(host) {
        // Lean on the synced leksihjelp i18n surface already loaded by
        // leksihjelp-loader.js (provides self.__lexiI18n with t, getUiLanguage,
        // langName). If not present (loader not yet booted), fall back to
        // identity strings so we don't crash before bundle load.
        const i18n = host.self?.__lexiI18n || {
            t: (k) => k,
            getUiLanguage: () => 'nb',
            setUiLanguage: () => {},
            langName: (c) => c,
        };
        return i18n;
    }

    /**
     * mountLeksihjelpSidepanel — call once per writing-environment lifecycle.
     * @param {{ rootEl: HTMLElement }} options
     * @returns {{ destroy: () => void }}
     */
    function mountLeksihjelpSidepanel({ rootEl }) {
        const host = window;
        const storage = makeStorageAdapter(host);
        const runtime = makeRuntimeAdapter(host);
        const i18n = makeI18n(host);

        // Build the two-pane sidepanel layout: dictionary on top (search-heavy),
        // settings collapsed below behind a small toggle. Vertical scroll so the
        // narrow sidepanel works.
        // EKSAMENMODUS badge: visible reinforcement that the dictionary is
        // exam-restricted. Reads chrome.storage.local.examMode (set by
        // writing-environment.js applyExamModeLock from Phase 29-02). Subscribes
        // to onChanged so the badge appears/disappears when the teacher flips
        // the profile mid-test.
        rootEl.innerHTML = `
            <div class="lh-sidepanel-stack" style="display:flex; flex-direction:column; height:100%; gap:8px;">
              <div id="lh-sidepanel-badge-row" style="flex:0 0 auto;">
                <span id="lh-sidepanel-exam-badge" class="exam-mode-badge" hidden>${i18n.t('examModeBadge') || 'EKSAMENMODUS'}</span>
              </div>
              <section id="lh-sidepanel-dictionary" class="view active" style="flex:1; min-height:0; overflow:auto;"></section>
              <details id="lh-sidepanel-settings-details" style="flex:0 0 auto;">
                <summary style="cursor:pointer; padding:6px 8px;">${i18n.t('settings_title') || 'Innstillinger'}</summary>
                <section id="lh-sidepanel-settings" class="view" style="max-height:40vh; overflow:auto;"></section>
              </details>
            </div>
        `;

        const examBadgeEl = rootEl.querySelector('#lh-sidepanel-exam-badge');
        const updateExamBadge = (on) => { examBadgeEl.hidden = !on; };
        host.chrome?.storage?.local?.get?.(['examMode'], (res) => updateExamBadge(!!res?.examMode));
        const examChangeListener = (changes) => {
            if ('examMode' in changes) updateExamBadge(!!changes.examMode.newValue);
        };
        host.chrome?.storage?.onChanged?.addListener?.(examChangeListener);

        const dictContainer = rootEl.querySelector('#lh-sidepanel-dictionary');
        const settingsContainer = rootEl.querySelector('#lh-sidepanel-settings');

        const handles = [];

        if (host.__lexiDictionaryView?.mount && host.__lexiVocab) {
            handles.push(host.__lexiDictionaryView.mount(dictContainer, {
                vocab: host.__lexiVocab,
                storage,
                runtime,
                t: i18n.t,
                getUiLanguage: i18n.getUiLanguage,
                langName: i18n.langName,
                isFeatureEnabled: () => true,
                getAllowedPronouns: () => null,
                audioEnabled: false,        // <-- KEY: no audio in lockdown
                playAudio: () => {},
                BACKEND_URL: '',
                onSearch: null,
            }));
        } else {
            dictContainer.innerHTML = `<p style="padding:12px;color:#888;">Leksihjelp er ikke klar enda.</p>`;
        }

        if (host.__lexiSettingsView?.mount) {
            handles.push(host.__lexiSettingsView.mount(settingsContainer, {
                storage,
                runtime,
                t: i18n.t,
                getUiLanguage: i18n.getUiLanguage,
                setUiLanguage: i18n.setUiLanguage,
                langName: i18n.langName,
                loadGrammarFeatures: async (lang) => {
                    // Lockdown ships grammar-features-{lang}.json in /public/leksihjelp/data/
                    const url = `/leksihjelp/data/grammarfeatures-${lang}.json`;
                    const res = await fetch(url);
                    return res.ok ? res.json() : null;
                },
                saveAndNotifyGrammarChange: () => runtime.sendMessage({ type: 'GRAMMAR_FEATURES_CHANGED' }),
                showSection: {
                    language: false,
                    grammar: true,
                    darkmode: true,
                    prediction: false,
                    spellcheckAlternates: false,
                },
            }));
        }

        return {
            destroy() {
                host.chrome?.storage?.onChanged?.removeListener?.(examChangeListener);
                handles.forEach(h => { try { h.destroy?.(); } catch (e) { /* non-fatal */ } });
                rootEl.innerHTML = '';
            },
        };
    }

    window.__leksihjelpSidepanelHost = { mount: mountLeksihjelpSidepanel };
})();
```
