`extension/exam-registry.js` exposes `host.__lexiExamRegistry` as a frozen array of `{ id, exam }`. Surface IDs (Plan 01):

Helper to introduce in `extension/content/spell-check-core.js` (single source of truth, used by all consumers via `host.__lexiExam`):
```js
host.__lexiExam = host.__lexiExam || {
  isSurfaceSafe(surfaceId, examMode) {
    if (!examMode) return true;
    const reg = host.__lexiExamRegistry || [];
    const e = reg.find(x => x.id === surfaceId);
    return e ? e.exam.safe === true : false; // unknown surface = fail-safe (hidden)
  },
  isRuleSafe(rule, examMode) {
    if (!examMode) return true;
    return rule && rule.exam && rule.exam.safe === true;
  },
  isExplainSafe(rule, examMode) {
    if (!examMode) return true;
    if (!rule || !rule.explain || !rule.explain.exam) return this.isRuleSafe(rule, examMode);
    return rule.explain.exam.safe === true;
  }
};
```
